More case for post order.
When tracing back, if the popped item has a right child and the right child is not processed yet, then make sure right child is processed before root

def postOrder(root):
    res=[]
    stack=[]
    preNode=None
    while root or stack:
        if root:
            stack.append(root)
            root=root.Left
        else:
            peekNode=stack[-1]
            #the peek node has a right child that haven't been processed
            if peekNode.right and preNode!=peekNode.right: 
                root=peekNode.right
            else:
                stack.pop()
                res.append(peekNode.val)
                preNode=peekNode
    return res
